# 3. シーケンス図

## 3.1. 初期化シーケンス

~~~plantuml

@startuml obniz初期化シーケンス

hide footbox

participant "HardwareModel\n（ハードウェアモデルクラス）" as HardwareModel
participant "Obniz" as Obniz
participant PeripheralI2C
participant "PCA9685\n（サーボモータードライバIC）" as PCA9685

activate HardwareModel
create Obniz
HardwareModel -> Obniz : new()
activate Obniz
    HardwareModel <-- Obniz : obnize
deactivate Obniz

    HardwareModel --> Obniz : onconnect()
    activate Obniz

        Obniz -> Obniz : getFreeI2C() : PeripheralI2C
        activate Obniz
            Obniz --> Obniz : i2c
        deactivate Obniz

        Obniz -> PeripheralI2C : start()
        activate PeripheralI2C            
        deactivate  PeripheralI2C

        Obniz -> Obniz : wired() : PCA9685
        activate Obniz
            create PCA9685
            Obniz -> PCA9685 : new ()
            activate PCA9685
                Obniz --> PCA9685
            deactivate PCA9685

            Obniz --> Obniz : driver
        deactivate Obniz

        note right of Obniz 
            接続表示
        endnote
        Obniz -> Display : print
        activate Display
        deactivate Display



    deactivate Obniz

deactivate Obniz

deactivate HardwareModel

@enduml

~~~

## 3.2. 方向キー入力シーケンス

~~~plantuml

@startuml

hide footbox

actor User
participant MainScene
participant document

User --> MainScene : 方向キー入力

    document -> MainScene :update()
    activate MainScene
        MainScene -> InputManager : checkDirection()
        activate InputManager
            MainScene <-- InputManager : dir
        deactivate InputManager

        alt 上方向キーが押されている
            note over of MainScene
                up++
            endnote
        else 左方向キーが押されている
            note over of MainScene
                left++
            endnote
        else 右方向キーが押されている
            note over of MainScene
                right++
            endnote
        else 下方向キーが押されている
            note over of MainScene
                down++
            endnote
        end
    deactivate MainScene

User --> MainScene : 方向キーを離す

    document -> MainScene : update()
    activate MainScene
        MainScene -> InputManager : checkButton()
        activate InputManager
            MainScene <-- InputManager : 
        deactivate InputManager

        alt 上方向キーが離された
            MainScene -> MainScene : onButtonRelease()
            activate MainScene
                MainScene --> RobotArmsController : armXdrive()
                activate RobotArmsController
                    RobotArmsController -> PCA9685 : drive()
                    activate PCA9685
                    deactivate PCA9685
                deactivate RobotArmsController
                MainScene -> MainScene : resetDirection()
                activate MainScene
                deactivate MainScene
            deactivate MainScene
        else 左方向キーが離された。
            MainScene -> MainScene : onButtonRelease()
            activate MainScene
                MainScene --> RobotArmsController : armXdrive()
                activate RobotArmsController
                    RobotArmsController -> PCA9685 : drive()
                    activate PCA9685
                    deactivate PCA9685
                deactivate RobotArmsController
                MainScene -> MainScene : resetDirection()
                activate MainScene
                deactivate MainScene
            deactivate MainScene
        else 右方向キーが離された。
            MainScene -> MainScene : onButtonRelease()
            activate MainScene
                MainScene --> RobotArmsController : armXdrive()
                activate RobotArmsController
                    RobotArmsController -> PCA9685 : drive()
                    activate PCA9685
                    deactivate PCA9685
                deactivate RobotArmsController
                MainScene -> MainScene : resetDirection()
                activate MainScene
                deactivate MainScene
            deactivate MainScene
        else 下方向キーが離された。
            MainScene -> MainScene : onButtonRelease()
            activate MainScene
                MainScene --> RobotArmsController : armXdrive()
                activate RobotArmsController
                    RobotArmsController -> PCA9685 : drive()
                    activate PCA9685
                    deactivate PCA9685
                deactivate RobotArmsController
                MainScene -> MainScene : resetDirection()
                activate MainScene
                deactivate MainScene
            deactivate MainScene
        end

    deactivate MainScene
@enduml
~~~
